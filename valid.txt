@echo off
chcp 65001 >nul
setlocal enabledelayedexpansion

python csv2dart.py

@if errorlevel 1 (
    echo Ошибка при выполнении csv2dart.py
    pause
    exit /b 1
)

echo Обработка файлов в папке MIH-BIN-DART

@if not exist "MIH-BIN-DART" (
    echo Папка MIH-BIN-DART не найдена!
    pause
    exit /b 1
)

@for %%f in ("MIH-BIN-DART\*") do (
    @if not "%%~xf"=="" (
        @dart run pan-tompkins-alg.dart "MIH-BIN-DART\%%~nxf" >nul 2>&1
	@echo Файл %%f обработан
    )
)

echo Все операции завершены
pause
